all x (CollectionOfStatements(x) -> InformationContentEntity(x)).

%Any instance of CollectionOfStatements is an instance of InformationContentEntity

all x(Argument(x) -> CollectionOfStatements(x)).

%Any instance of Argument is an instance of CollectionOfStatements

all x(Statement(x) -> InformationContentEntity(x)).

%Any instance of Statement is an instance of InformationContentEntity

-(exists x(Statement(x) & CollectionOfStatements(x))).

%Statement and CollectionOfStatements are disjoint

all x(Premise(x) -> (Statement(x) & -(Conclusion(x) | Supposition(x)))).

%Any instance of Premise is an instance of Statement, disjoint from Conclusion and Supposition

all x(Conclusion(x) -> (Statement(x) & -(Premise(x) | Supposition(x)))).

%Any instance of Conclusion is an instance of Statement, disjoint from Premise and Supposition

all x(Supposition(x) -> (Statement(x) & -(Conclusion(x) | Premise(x)))).

%Any instance of Supposition is an instance of Statement, disjoint from Conclusion and Premise

all x(Subconclusion(x) -> Statement(x)).

%Any instance of Subconcusion is an instance of Statement

all x (LanguageAct(x) -> IntentionalAct(x)).

%Any instance of LanguageAct is an instance of IntentionalAct

all x (ActOfArguing(x) -> (LanguageAct(x) & -(ActOfArgumentCreation(x) | ActOfSupposing(x) | ActOfAffirming | ActOfInferring(x)))).

%Any instance of ActOfArguing is an instance of LanguageAct, disjoint from ActOfArgumentCreation, ActOfSupposing, ActOfAffirming, and ActOfInferring

all x(ActOfArgumentCreation(x) -> (LanguageAct(x) & -(ActOfArguing(x) | ActOfSupposing(x) | ActOfAffirming | ActOfInferring(x)))).

%Any instance of ActOfArgumentCreation is an instance of LanguageAct, disjoint from ActOfArguing, ActOfSupposing, ActOfAffirming, and ActOfInferring

all x(ActOfSupposing(x) -> (LanguageAct(x) & -(ActOfArgumentCreation(x) | ActOfArguing(x) | ActOfAffirming | ActOfInferring(x)))).

%Any instance of ActOfSupposing is an instance of LanguageAct, disjoint from ActOfArguing, ActOfArgumentCreation, ActOfAffirming, and ActOfInferring

all x(ActOfInferring(x) -> (LanguageAct(x) & -(ActOfArgumentCreation(x) | ActOfSupposing(x) | ActOfAffirming | ActOfArguing(x)))).

%Any instance of ActOfInferring is an instance of LanguageAct, disjoint from ActOfArguing, ActOfArgumentCreation, ActOfAffirming, and ActOfSupposing

all x(ActOfAffirming(x) -> (LanguageAct(x) & -(ActOfArgumentCreation(x) | ActOfSupposing(x) | ActOfInferring | ActOfArguing(x)))).

%Any instance of ActOfAffirming is an instance of LanguageAct, disjoint from ActOfArguing, ActOfArgumentCreation, ActOfInferring, and ActOfSupposing
